ðŸ“Š Charting with matplotlib

import matplotlib.pyplot as plt
from datetime import datetime

dates = ["2025-09-01", "2025-09-03", "2025-09-05"]
pages = [15, 35, 50]
dates = [datetime.strptime(d, "%Y-%m-%d") for d in dates]

plt.plot(dates, pages, marker='o', label="Your Progress")
plt.xlabel("Date")
plt.ylabel("Pages Read")
plt.title("Reading Progress")
plt.legend()
plt.grid(True)
plt.show()
